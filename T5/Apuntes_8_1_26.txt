		atributos -> cualifican al objeto. private nombre;
		constructores -> mínimo -> 1 -> con firmas diferentes -> public NombreClase(){}
		metodos -> funcionalidad al futuro objeto -> con firmas diferentes -> public void nombreMetodo(parámetros){}

		main -> Alumno borja = new Alumno()

HERENCIA -> obtener "algo" de alguien.
			Padres -> Hijo -> Hijos
			Solo se puede heredar de una clase al mismo tiempo
			En herencia no se heredan 
				los constructores -> heredarlos manualmente -> mediante la palabra reservada super() -> en la primera linea
				las variables o metodos privados -> no se heredan
			Superclase: clase de la cual hereda otra clase, utilizando una palabra reservada extends
